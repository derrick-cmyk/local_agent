```python
import tkinter as tk

class TicTacToe:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Tic Tac Toe")
        
        # Create the board grid
        self.board = [['' for _ in range(3)] for _ in range(3)]
        self.buttons = [[tk.Button(self.root, text="", height=5, width=10, command=lambda row=i, col=j: self.on_button_click(row, col)) for j in range(3)] for i in range(3)]
        
        # Place the buttons on the grid
        for i in range(3):
            for j in range(3):
                self.buttons[i][j].grid(row=i, column=j)
        
        self.root.mainloop()

    def on_button_click(self, row, col):
        if self.board[row][col] == '':
            self.board[row][col] = 'X'
            self.buttons[row][col]['text'] = 'X'
            if self.check_winner() or self.is_full():
                return
            self.ai_move()
    
    def ai_move(self):
        # Simple AI to randomly select an empty space
        for i in range(3):
            for j in range(3):
                if self.board[i][j] == '':
                    self.board[i][j] = 'O'
                    self.buttons[i][j]['text'] = 'O'
                    return
    
    def check_winner(self):
        # Check rows, columns, and diagonals
        for i in range(3):
            if all(self.board[i][j] == 'X' for j in range(3)) or all(self.board[j][i] == 'X' for j in range(3)):
                return True
        if self.board[0][0] == 'X' and self.board[1][1] == 'X' and self.board[2][2] == 'X':
            return True
        if self.board[0][2] == 'X' and self.board[1][1] == 'X' and self.board[2][0] == 'X':
            return True
        return False
    
    def is_full(self):
        # Check if the board is full
        for i in range(3):
            for j in range(3):
                if self.board[i][j] == '':
                    return False
        return True

if __name__ == "__main__":
    TicTacToe()
```

This code creates a simple text-based interface for a Tic Tac Toe game. It includes a 3x3 grid of buttons, and the `on_button_click` function handles player moves by placing an 'X' on the clicked button. The AI places an 'O' randomly in an empty space after each human move. The game checks for a winner or if the board is full to end the game.
